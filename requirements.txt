# =============================================================================
# requirements.txt - Python dependencies voor Ansible Windows beheer
# =============================================================================
# Installeren met: pip install -r requirements.txt
#
# Aanbevolen: gebruik een Python virtual environment
#   python3 -m venv .venv
#   source .venv/bin/activate
#   pip install -r requirements.txt

# --- Ansible core -----------------------------------------------------------
# Minimaal versie 2.16 vereist voor Windows Server 2022 ondersteuning
ansible-core>=2.16,<2.20

# --- Ansible lint -----------------------------------------------------------
ansible-lint>=24.0

# --- PSRP verbinding --------------------------------------------------------
# pypsrp: Python library voor PSRP (PowerShell Remoting Protocol).
# Sneller dan pywinrm omdat PSRP pipelining ondersteunt (pywinrm niet):
# taken hergebruiken dezelfde verbinding in plaats van per taak opnieuw verbinden.
# Basic auth over HTTP does not require NTLM; pypsrp 0.9.x no longer declares
# the [ntlm] extra so omit it to silence the pip warning.
pypsrp>=0.7.0

# requests: HTTP library (dependency van pypsrp)
requests>=2.31.0

# Kerberos authenticatie (aanbevolen voor Active Directory omgevingen)
# Uncomment als Kerberos gebruikt wordt en installeer ook:
#   sudo apt-get install python3-dev libkrb5-dev krb5-user  (Ubuntu/Debian)
#   sudo yum install python3-devel krb5-devel krb5-workstation  (RHEL/CentOS)
# pypsrp[kerberos]

# --- Ansible Vault ----------------------------------------------------------
# Cryptography library voor Ansible Vault encryptie
cryptography>=41.0.0

# Jinja2: templating engine (wordt meegeleverd met ansible-core, maar
# expliciet vermeld voor duidelijkheid)
Jinja2>=3.1.0
